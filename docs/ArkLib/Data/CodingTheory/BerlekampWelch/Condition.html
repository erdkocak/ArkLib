    Prop
    • cond (i : Fin n) : Polynomial.eval (ωs i) Q = f i * Polynomial.eval (ωs i) E
    • E_natDegree : E.natDegree = e
    • E_leadingCoeff : E.coeff e = 1
    • Q_natDegree : Q.natDegree ≤ e + k - 1
